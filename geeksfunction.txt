#Check even or odd function calling
def func1(num):
    if num % 2 == 0:
        return f"{num} is Even and true"
    else:
        return f"{num} is Odd and false"
print(func1(10))


def even1(n):
    return n % 2 == 0
print(even1(7))

#map
